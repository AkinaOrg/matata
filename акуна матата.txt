# Запитуємо користувача про кількість курей, свиней і корів
try:
    kury = int(input("Введіть кількість курей: "))
    svini = int(input("Введіть кількість свиней: "))
    korovi = int(input("Введіть кількість корів: "))
except ValueError:
    print("Будь ласка, введіть цілі числа для кількості тварин.")
    exit()

# Розраховуємо загальну кількість ніг
kilka_nig = (kury * 2) + (svini * 4) + (korovi * 4)

# Виводимо результат на консоль
print(f"Загальна кількість ніг на фермі: {kilka_nig} ніг")
